fn first_word(s: &String) -> usize {
  let bytes = s.as_bytes(); // 变成 [u8] 数组的引用了
  // 先取迭代器，然后使用 enumerate() 包装一次，最后模式匹配（解构赋值）成索引 i 和数据引用 &item 的元组
  for (i, &item) in bytes.iter().enumerate()  {
  	if item == b' ' {
      return i;
    }
  }
  return s.len();
}
